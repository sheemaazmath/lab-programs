import numpy as np
import matplotlib.pyplot as plt
from collections import Counter
data = np.random.rand(100)
train, test = data[:50], data[50:]
labels = ["Class1" if x <= 0.5 else "Class2" for x in train]
def knn(x, k):
    d = sorted((abs(x - train[i]), labels[i]) for i in range(50))
    return Counter(l for _, l in d[:k]).most_common(1)[0][0]
print("--- k-Nearest Neighbors Classification ---")
print("Training dataset: First 50 points labeled based on the rule (x <= 0.5 -> Class1, x > 0.5 -> Class2)")
print("Testing dataset: Remaining 50 points to be classified\n")
results = {}
for k in [1, 2, 3, 4, 5, 20, 30]:
    print(f"Results for k = {k}:")
    preds = [knn(x, k) for x in test]
    results[k] = preds
    for i, (x, p) in enumerate(zip(test, preds), 51):
        print(f"Point x{i} (value: {x:.4f}) is classified as {p}")
    print("\n")
print("Classification complete.\n")
for k, preds in results.items():
    c1 = [test[i] for i in range(50) if preds[i] == "Class1"]
    c2 = [test[i] for i in range(50) if preds[i] == "Class2"]

    plt.figure(figsize=(10, 6))
    plt.scatter(train, [0]*50, c=["blue" if l=="Class1" else "red" for l in labels], label="Training Data")
    plt.scatter(c1, [1]*len(c1), c="blue", marker="x", label="Class1 (Test)")
    plt.scatter(c2, [1]*len(c2), c="red", marker="x", label="Class2 (Test)")
    plt.title(f"k-NN Classification Results for k = {k}")
    plt.xlabel("Data Points")
    plt.ylabel("Classification Level")
    plt.legend()
    plt.grid(True)
    plt.show()

import numpy as np
import matplotlib.pyplot as plt

def lwr(x, X, y, tau):
    w = np.exp(-np.sum((X - x)**2, axis=1) / (2 * tau**2))
    Xw = X * w[:, None]
    theta = np.linalg.inv(X.T @ Xw) @ (X.T @ (w * y))
    return x @ theta
np.random.seed(42)
X = np.linspace(0, 2*np.pi, 100)
y = np.sin(X) + 0.1*np.random.randn(100)
Xb = np.c_[np.ones(len(X)), X]
xt = np.linspace(0, 2*np.pi, 200)
xtb = np.c_[np.ones(len(xt)), xt]
y_pred = np.array([lwr(xi, Xb, y, 0.5) for xi in xtb])
plt.scatter(X, y, c='r', alpha=0.7)
plt.plot(xt, y_pred, 'b', lw=2)
plt.title("Locally Weighted Regression")
plt.xlabel("X"); plt.ylabel("y")
plt.grid(alpha=0.3)
plt.show()

import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import fetch_olivetti_faces as load
from sklearn.model_selection import train_test_split as split, cross_val_score
from sklearn.naive_bayes import GaussianNB as GNB
from sklearn.metrics import accuracy_score as acc, classification_report as report, confusion_matrix as cm

X, y = load(shuffle=True, random_state=42).data, load(shuffle=True, random_state=42).target
Xtr, Xte, ytr, yte = split(X, y, test_size=0.3, random_state=42)
model = GNB().fit(Xtr, ytr)
pred = model.predict(Xte)
print(f'Accuracy: {acc(yte, pred)*100:.2f}%')
print("\nClassification Report:\n", report(yte, pred, zero_division=1))
print("\nConfusion Matrix:\n", cm(yte, pred))
cv = cross_val_score(model, X, y, cv=5, scoring='accuracy')
print(f'\nCross-validation accuracy: {cv.mean()*100:.2f}%')
fig, ax = plt.subplots(3, 5, figsize=(12, 8))
for a, img, t, p in zip(ax.ravel(), Xte, yte, pred):
    a.imshow(img.reshape(64, 64), cmap='gray')
    a.set_title(f"True: {t}, Pred: {p}")
    a.axis('off')
plt.show()

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from sklearn.datasets import load_breast_cancer
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA
from sklearn.metrics import confusion_matrix, classification_report
X, y = load_breast_cancer(return_X_y=True)
X = StandardScaler().fit_transform(X)
kmeans = KMeans(n_clusters=2, random_state=42)
labels = kmeans.fit_predict(X)
print("Confusion Matrix:")
print(confusion_matrix(y, labels))
print("\nClassification Report:")
print(classification_report(y, labels))
X_pca = PCA(2).fit_transform(X)
df = pd.DataFrame(X_pca, columns=["PC1", "PC2"])
df["Cluster"] = labels
df["True Label"] = y
centers = PCA(2).fit(X).transform(kmeans.cluster_centers_)
plt.figure(figsize=(8,6))
sns.scatterplot(data=df, x="PC1", y="PC2", hue="Cluster",
                palette="Set1", s=100, edgecolor="black", alpha=0.7)
plt.title("K-Means Clustering of Breast Cancer Dataset")
plt.show()
plt.figure(figsize=(8,6))
sns.scatterplot(data=df, x="PC1", y="PC2", hue="True Label",
                palette="coolwarm", s=100, edgecolor="black", alpha=0.7)
plt.title("True Labels of Breast Cancer Dataset")
plt.show()
plt.figure(figsize=(8,6))
sns.scatterplot(data=df, x="PC1", y="PC2", hue="Cluster",
                palette="Set1", s=100, edgecolor="black", alpha=0.7)
plt.scatter(centers[:,0], centers[:,1], s=200, c="red", marker="X", label="Centroids")
plt.title("K-Means Clustering with Centroids")
plt.legend()
plt.show()

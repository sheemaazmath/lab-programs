    r = model.most_similar(positive=["king", "woman"], negative=["man"])

    print(f"Error: {e}")

    sim = model.similarity("king", "queen")
    print(f"\nSimilarity between 'king' and 'queen': {sim:.4f}")
except Exception as e:
    print(f"Error: {e}")
try:
    odd = model.doesnt_match(["apple", "banana", "grape", "car"])

import gensim.downloader as api, numpy as np, matplotlib.pyplot as plt
from sklearn.decomposition import PCA
print("Loading model... (This may take a while)")

print("Model loaded!")

import nltk, matplotlib.pyplot as plt
from gensim.models import Word2Vec
from nltk.tokenize import word_tokenize
from sklearn.manifold import TSNE
c = [
"The doctor diagnosed the patient with diabetes.",
"Insulin is used to treat diabetes.",
"A cardiologist specializes in heart diseases.",
"Patients with hypertension should reduce salt intake.",
"Antibiotics treat bacterial infections.",
"Vaccines help build immunity.",
"Surgery removes tumors.",
"A neurologist treats nervous system disorders."
]
nltk.download('punkt')
m = Word2Vec([word_tokenize(i.lower()) for i in c], vector_size=50, window=5, min_count=1, workers=4)
m.save('medical_word2vec.model')
w = list(m.wv.index_to_key)
rv = TSNE(n_components=2, random_state=42).fit_transform(m.wv[w])
plt.figure(figsize=(8,5))
for i,wd in enumerate(w):
    x,y = rv[i]; plt.scatter(x,y); plt.annotate(wd,(x,y),fontsize=10)
plt.title("Word Embeddings Visualization"); plt.show()

try:
    print("Words similar to 'diabetes':")
    for s,sc in m.wv.most_similar("diabetes",topn=5):
        print(f"{s} (Similarity: {sc:.2f})")
except:
    print("'diabetes' not found in vocabulary")

from pydantic import BaseModel
import wikipediaapi
class I(BaseModel):
    name:str; founder:str; founded_year:str; branches:str; employees:str; summary:str

n = input("Enter Institution Name: ")
w = wikipediaapi.Wikipedia(user_agent="MyWikipediaScraper/1.0 (contact: myemail@example.com)", language="en")
p = w.page(n)
try:
    if not p.exists(): raise ValueError("Institution page does not exist on Wikipedia")

    s = " ".join(p.summary.split(".")[:4]) + "."
    f = y = b = e = "Not Available"

    for sec in p.sections:
        t = sec.title.lower(); txt = sec.text.split(". ")[0]
        if "founder" in t: f = txt
        if "founded" in t: y = txt
        if "branches" in t: b = txt
        if "employees" in t: e = txt

    print(I(name=n, founder=f, founded_year=y, branches=b, employees=e, summary=s).model_dump_json(indent=4))

except ValueError as err:
    print(err)

import PyPDF2, re
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
# Load local PDF
try:
    t="".join([(p.extract_text() or "") for p in PyPDF2.PdfReader(open("ipc.pdf","rb")).pages])
    print("PDF loaded successfully!")
except Exception as e:
    print(f"Error: {e}")
    t=""
# Create index
d={}
if t:
    for m in re.findall(r"((?:CHAPTER|SECTION)\s+\w+.*?)(?=(?:CHAPTER|SECTION)\s+\w+|$)",t,re.S|re.I):
        x=re.search(r"^(?:CHAPTER|SECTION)\s+\w+\.?\s+(.*?)(?=\n)",m,re.S|re.I)
        if x: d[x.group(1).strip()]=m[x.end():].strip()
# Chatbot
def resp(q):
    if not d: return "I'm sorry, I cannot process the IPC without the index."
    s=list(d.values()); k=list(d.keys())
    try:
        v=TfidfVectorizer().fit_transform(s+[q])
        sim=cosine_similarity(v[-1],v[:-1]).flatten()
        if not sim.any(): return "I'm sorry, I couldn't find relevant information."
        key=k[sim.argmax()]
        txt=re.sub(r'\n+','\n',re.sub(r'[ \t]+\n','\n',re.sub(r'\n\s*\n+','\n\n',d[key].strip())))
        return f"Here's what I found in the Indian Penal Code, **{key}**:\n\n{txt}"
    except:
        return "I'm sorry, I couldn't process your query."

print("Welcome to the Indian Penal Code Chatbot! Type 'exit' to quit.")
while 1:
    q=input("You: ")
    if q.lower()=="exit": break
    print(f"Chatbot: {resp(q)}\n")
